Enunciado

Un centro logístico utiliza una matriz de 3x3 para monitorizar el stock de 9 estanterías de
un almacén automatizado. Cada celda de la matriz contiene un número entero que
representa la cantidad de unidades de un producto. El sistema debe determinar el estado
del almacén según los siguientes criterios, ordenados de más prioritario a menos (sólo se
mostrará el primero que se cumpla):

●​ STOCK OUT: Si alguna de las estanterías tiene 0 unidades.
●​ ZONE WARNING: Si la suma de unidades de cualquiera de las tres filas (Zonas)
es inferior a 15.
●​ LOW INVENTORY: Si la suma total de todas las unidades del almacén es inferior
a 100.
●​ OPERATIONAL: Si no se cumple ninguna de las condiciones anteriores.

Se pide implementar un programa en C que:

1.​ Lea el nombre de un fichero de texto del teclado.
2.​ Cargue los números en una matriz de 3x3.
3.​ Verifique el estado del almacén según los criterios descritos.

Entrada de datos
Se introducirá por teclado el nombre del archivo (máximo 15 caracteres) que contiene 9
números enteros organizados en formato de matriz 3x3.

Salida de datos
Se mostrará el estado jerárquico del almacén según los literales indicados.
Ejemplo de ejecución

En el anexo del final de este documento podéis consultar un ejemplo de ejecución del
programa en lenguaje C: revisadlo para utilizar los literales exactos para resolver el
ejercicio. Tened en cuenta que la corrección se hace de manera automática.

algorithm inventoryManagement

    const
        MAX_LEN: integer = 16;
        ROWS: integer = 3;
        COLS: integer = 3;
        STOCK_OUT: integer = 0;
        ZONE_WARNING: integer = 15;
        LOW_INVENTORY: integer = 100;
        STATE_STOCK_OUT: integer = 0;
        STATE_ZONE_WARNING: integer = 1;
        STATE_LOW_INVENTORY: integer = 2;
        STATE_OPERATIONAL: integer = 3;
    end const

    var
        filename: string;
        stock: matrix[ROWS][COLS] of integer;
        file: file Pointer;  { O simplemente file de tipo fichero }
        zone0Total: integer;
        zone1Total: integer;
        zone2Total: integer;
        total: integer;
        state: integer;
    end var

    { Input section }
    writeString("INPUT");
    writeString("FILENAME?");
    filename := readString();

    file := openFile(filename, "r");
    
    { Lectura de la matriz desde fichero }
    stock[1][1] := readIntegerFromFile(file);
    stock[1][2] := readIntegerFromFile(file);
    stock[1][3] := readIntegerFromFile(file);
    stock[2][1] := readIntegerFromFile(file);
    stock[2][2] := readIntegerFromFile(file);
    stock[2][3] := readIntegerFromFile(file);
    stock[3][1] := readIntegerFromFile(file);
    stock[3][2] := readIntegerFromFile(file);
    stock[3][3] := readIntegerFromFile(file);
    
    closeFile(file);

    { Processing section }
    zone0Total := stock[1][1] + stock[1][2] + stock[1][3];
    zone1Total := stock[2][1] + stock[2][2] + stock[2][3];
    zone2Total := stock[3][1] + stock[3][2] + stock[3][3];
    
    total := zone0Total + zone1Total + zone2Total;

    if (stock[1][1] = STOCK_OUT or stock[1][2] = STOCK_OUT or stock[1][3] = STOCK_OUT or
        stock[2][1] = STOCK_OUT or stock[2][2] = STOCK_OUT or stock[2][3] = STOCK_OUT or
        stock[3][1] = STOCK_OUT or stock[3][2] = STOCK_OUT or stock[3][3] = STOCK_OUT) then
        
        state := STATE_STOCK_OUT;
        
    else if (zone0Total < ZONE_WARNING or zone1Total < ZONE_WARNING or zone2Total < ZONE_WARNING) then
        
        state := STATE_ZONE_WARNING;
        
    else if (total < LOW_INVENTORY) then
        
        state := STATE_LOW_INVENTORY;
        
    else
        
        state := STATE_OPERATIONAL;
        
    end if

    { Output section }
    writeString("OUTPUT");

    switch state do
        case STATE_STOCK_OUT:
            writeString("STOCK OUT");
        case STATE_ZONE_WARNING:
            writeString("ZONE WARNING");
        case STATE_LOW_INVENTORY:
            writeString("LOW INVENTORY");
        default:
            writeString("OPERATIONAL");
    end switch

end algorithm

/* Description: This program reads stock data from a file, stores it in a
matrix, and processes the information to display the results. */

#include <stdio.h>

/* constants */
#define MAX_LEN (15 + 1)
#define ROWS 3
#define COLS 3
#define STOCK_OUT 0
#define ZONE_WARNING 15
#define LOW_INVENTORY 100
#define STATE_STOCK_OUT 0
#define STATE_ZONE_WARNING 1
#define STATE_LOW_INVENTORY 2
#define STATE_OPERATIONAL 3

int main(void) {
/* ----------------------- Variable declarations ---------------------------- */
    char filename[MAX_LEN];
    int stock[ROWS][COLS];
    FILE *file;
    int zone0Total, zone1Total, zone2Total, total, state;

/* ---------------------------- Input section ------------------------------- */
    printf("INPUT\n");

    printf("FILENAME?\n");
    scanf("%15s", filename);
    file = fopen(filename, "r");

    fscanf(file, "%d %d %d", &stock[0][0], &stock[0][1], &stock[0][2]);
    fscanf(file, "%d %d %d", &stock[1][0], &stock[1][1], &stock[1][2]);
    fscanf(file, "%d %d %d", &stock[2][0], &stock[2][1], &stock[2][2]);

    fclose(file);

/* ------------------------- Processing section ----------------------------- */

    /* calculate total zone */
    zone0Total = stock[0][0] + stock[0][1] + stock[0][2];
    zone1Total = stock[1][0] + stock[1][1] + stock[1][2];
    zone2Total = stock[2][0] + stock[2][1] + stock[2][2];

    /* calculate total stock */
    total = zone0Total + zone1Total + zone2Total;

    /* STOCK OUT */
    if (stock[0][0] == STOCK_OUT ||
        stock[0][1] == STOCK_OUT ||
        stock[0][2] == STOCK_OUT ||
        stock[1][0] == STOCK_OUT ||
        stock[1][1] == STOCK_OUT ||
        stock[1][2] == STOCK_OUT ||
        stock[2][0] == STOCK_OUT ||
        stock[2][1] == STOCK_OUT ||
        stock[2][2] == STOCK_OUT) {

        state = STATE_STOCK_OUT;

    } else if (zone0Total < ZONE_WARNING ||
               zone1Total < ZONE_WARNING ||
               zone2Total < ZONE_WARNING) {

        state = STATE_ZONE_WARNING;

    } else if (total < LOW_INVENTORY) {

        state = STATE_LOW_INVENTORY;

    } else {

        state = STATE_OPERATIONAL;
    }

/* -------------------------- Output section -------------------------------- */
    printf("OUTPUT\n");

    switch (state) {

        case STATE_STOCK_OUT:
            printf("STOCK OUT\n");
            break;

        case STATE_ZONE_WARNING:
            printf("ZONE WARNING\n");
            break;

        case STATE_LOW_INVENTORY:
            printf("LOW INVENTORY\n");
            break;

        default:
            printf("OPERATIONAL\n");

    }
    return 0;
}
